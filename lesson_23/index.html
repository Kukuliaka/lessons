Задача №2

Написати функцію, яка додає в <body> список UL з певною кількістю LI

(кількість має передаватись як параметр функції, також мати значення за замовченням)

Відповідь: 
							
								function addUlToBody(n = 4) {
									const someList = document.createElement('ul')
									if (n === Number && n !== 0) return
									for (let i = 0; i < n; ++i) {
										someList.innerHTML += `<li>Пункт №${i + 1}</li>`
									}
									bodyElement.insertAdjacentElement("beforebegin", someList)
								}
								addUlToBody(8)
